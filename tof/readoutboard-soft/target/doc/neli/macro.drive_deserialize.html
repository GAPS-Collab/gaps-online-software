Expand description

This macro can be used to deserialize a single field in a struct.

Examples

use neli::err::DeError;

fn drive_deserialize() -> Result<(), DeError> {
    let vec = vec![1];
    assert_eq!(
        neli::drive_deserialize!(u8, vec.as_slice(), 0),
        (1u8, 1)
    );
    Ok(())
}

drive_deserialize().unwrap();
